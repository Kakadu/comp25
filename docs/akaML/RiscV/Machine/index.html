type reg =
  1. | Zero
    (*

    Always zero

    *)
  2. | RA
    (*

    Return address

    *)
  3. | SP
    (*

    Stack pointer

    *)
  4. | A of int
    (*

    Arguments A0..A7

    *)
  5. | T of int
    (*

    Temporary T0..T6

    *)
  6. | S of int
    (*

    Saved S0..S11

    *)

type instr =
  1. | Addi of reg * reg * int
    (*

    addi rd, rs, imm Adds register rs and immediate imm, result in rd

    *)
  2. | Add of reg * reg * reg
    (*

    add rd, rs1, rs2 Adds registers rs1 and rs2, result in rd

    *)
  3. | Sub of reg * reg * reg
    (*

    sub rd, rs1, rs2 Subtracts rs2 from rs1, result in rd

    *)
  4. | Mul of reg * reg * reg
    (*

    mul rd, rs1, rs2 Multiplies rs1 by rs2, result in rd

    *)
  5. | Xori of reg * reg * int
    (*

    xori rd, rs, imm Bitwise exclusive-or of rs and immediate imm, result in rd

    *)
  6. | Xor of reg * reg * reg
    (*

    xor rd, rs1, rs2 Bitwise exclusive-or of rs1 and rs2, result in rd

    *)
  7. | Slt of reg * reg * reg
    (*

    slt rd, rs1, rs2 Sets rd = 1 if rs1 < rs2, else rd = 0

    *)
  8. | Seqz of reg * reg
    (*

    seqz rd, rs Sets rd = 1 if rs == 0, else rd = 0

    *)
  9. | Snez of reg * reg
    (*

    snez rd, rs Sets rd = 1 if rs != 0, else rd = 0

    *)
  10. | Li of reg * int
    (*

    li rd, imm Loads immediate imm into register rd

    *)
  11. | La of reg * string
    (*

    la rd, label Loads the address of label into register rd

    *)
  12. | Mv of reg * reg
    (*

    mv rd, rs Copies the value from register rs into register rd

    *)
  13. | Ld of reg * offset
    (*

    ld rd, offset(base) Loads value from memory base + offset into register rd

    *)
  14. | Sd of reg * offset
    (*

    sd rs, offset(base) Stores value of register rs into memory base + offset

    *)
  15. | Beq of reg * reg * string
    (*

    beq rs1, rs2, label Jumps to label if rs1 == rs2

    *)
  16. | J of string
    (*

    j label Unconditional jump to label

    *)
  17. | Label of string
    (*

    label: Declares a label for jumps and function calls

    *)
  18. | Call of string
    (*

    call label Calls a function at label, return address stored in RA

    *)
  19. | Ret
    (*

    ret Returns from function to address stored in RA

    *)
  20. | Ecall
    (*

    ecall Environment call (system call). The syscall number is passed in A7, arguments in A0â€“A6, result in A0.

    *)
