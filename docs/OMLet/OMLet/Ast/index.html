  1. | Ident of string

  7. | Binary_add
  8. | Binary_subtract
  9. | Binary_multiply
  10. | Logical_or
  11. | Logical_and
  12. | Binary_divide
  13. | Binary_or_bitwise
  14. | Binary_xor_bitwise
  15. | Binary_and_bitwise
  16. | Binary_cons
val show_binary_operator : binary_operator -> string
type unary_operator =
  1. | Unary_minus
  2. | Unary_not
val show_unary_operator : unary_operator -> string
type pattern =
  1. | Wild
  2. | PList of pattern list
  3. | PCons of pattern * pattern
  4. | PTuple of pattern * pattern * pattern list
  5. | PConst of literal
  6. | PVar of ident
  7. | POption of pattern option

  1. | Nonrec

  2. | Tuple of expr * expr * expr list
  3. | List of expr list
  4. | Variable of ident

  10. | Function of case * case list
  11. | Match of expr * case * case list

  13. | Option of expr option
  14. | EConstraint of expr * TypedTree.typ
and let_bind =
  1. | Let_bind of pattern * pattern list * expr
val show_case : case -> string

  1. | Let of is_recursive * let_bind * let_bind list

  2. | Statement of statement
