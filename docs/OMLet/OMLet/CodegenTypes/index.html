type reg =
  1. | Zero
  2. | Ra
  3. | Fp
  4. | Sp
  5. | Stack of int * reg
  6. | Temp of int
  7. | Saved of int
  8. | Arg of int
type variable_type =
  1. | Argument
  2. | Local
type meta_info =
  1. | Var of int * variable_type
  2. | Func of string * int
  3. | Value of reg
val temp : int -> reg
val saved : int -> reg
val arg : int -> reg
type rtype_op =
  1. | ADD
  2. | SUB
  3. | MUL
  4. | DIV
  5. | AND
  6. | OR
  7. | XOR
  8. | SLT
type itype_op =
  1. | ADDI
  2. | JALR
  3. | SLTI
  4. | XORI
type stack_op =
  1. | LW
  2. | SW
  3. | LD
  4. | SD
type btype_op =
  1. | BEQ
  2. | BNE
  3. | BLE
type utype_op =
  1. | LUI
  2. | AUIPC
type jtype_op =
  1. | JAL
type pseudo_instr =
  1. | LI of reg * int
  2. | LA of reg * string
  3. | MV of reg * reg
  4. | J of string
  5. | RET
  6. | CALL of string
  7. | SNEZ of reg * reg
  8. | SEQZ of reg * reg
type true_instr =
  1. | RType of rtype_op * reg * reg * reg
  2. | IType of itype_op * reg * reg * int
  3. | StackType of stack_op * reg * reg
  4. | BType of btype_op * reg * reg * string
  5. | UType of utype_op * reg * int
  6. | JType of jtype_op * reg * int
  7. | Label of string
  8. | Ecall
type instr =
  1. | True of true_instr
  2. | Pseudo of pseudo_instr
val pp_reg : Stdlib.Format.formatter -> reg -> unit
val pp_rtype_op : Stdlib.Format.formatter -> rtype_op -> unit
val pp_itype_op : Stdlib.Format.formatter -> itype_op -> unit
val pp_stack_op : Stdlib.Format.formatter -> stack_op -> unit
val pp_btype_op : Stdlib.Format.formatter -> btype_op -> unit
val pp_utype_op : Stdlib.Format.formatter -> utype_op -> unit
val pp_jtype_op : Stdlib.Format.formatter -> jtype_op -> unit
val pp_pseudo_instr : Stdlib.Format.formatter -> pseudo_instr -> unit
val pp_true_instr : Stdlib.Format.formatter -> true_instr -> unit
val pp_instr : instr -> unit
