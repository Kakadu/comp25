type immexpr =
  1. | ImmNum of int
  2. | ImmId of Ast.ident
type cbinop =
  1. | CPlus
  2. | CMinus
  3. | CMul
  4. | CDiv
  5. | CEq
  6. | CNeq
  7. | CLt
  8. | CLte
  9. | CGt
  10. | CGte
type cexpr =
  1. | CBinop of cbinop * immexpr * immexpr
  2. | CIte of cexpr * aexpr * aexpr option
  3. | CImmexpr of immexpr
  4. | CLam of Ast.ident * aexpr
  5. | CApp of immexpr * immexpr list
and aexpr =
  1. | ALet of Ast.ident * cexpr * aexpr
  2. | ACExpr of cexpr
type aconstruction =
  1. | AExpr of aexpr
  2. | AStatement of Ast.is_recursive * (Ast.ident * aexpr) list
type aconstructions = aconstruction list
val count : int Stdlib.ref
val gen_temp : string -> Ast.ident
val binop_map : Ast.binary_operator -> string * cbinop
val collect_app_args : Ast.expr -> Ast.expr * Ast.expr list
val anf : Ast.expr -> (immexpr -> aexpr) -> aexpr
val anf_construction : Ast.construction -> aconstruction
val anf_constructions : Ast.construction list -> aconstructions
