type immexpr =
  1. | ImmNum of int
  2. | ImmId of Ast.ident
type cbinop =
  1. | CPlus
  2. | CMinus
  3. | CMul
  4. | CDiv
  5. | CEq
  6. | CNeq
  7. | CLt
  8. | CLte
  9. | CGt
  10. | CGte
type cexpr =
  1. | CBinop of cbinop * immexpr * immexpr
  2. | CIte of cexpr * aexpr * aexpr option
  3. | CImmexpr of immexpr
  4. | CLam of Ast.ident * aexpr
  5. | CApp of immexpr * immexpr list
and aexpr =
  1. | ALet of Ast.ident * cexpr * aexpr
  2. | ACExpr of cexpr
type aconstruction =
  1. | AExpr of aexpr
  2. | AStatement of Ast.is_recursive * (Ast.ident * aexpr) list
type aconstructions = aconstruction list
type anf_error =
  1. | Unreachable
  2. | Not_Yet_Implemented of string
type lifted_state = {
  1. lifted_lams : (Ast.ident * cexpr) list;
  2. lifted_letins : (Ast.ident * cexpr) list;
}
val gen_temp : string -> (Ast.ident, 'a) ResultCounter.ResultCounterMonad.t
val anf_constructions : lifted_state -> Ast.construction list -> (aconstruction list * lifted_state, anf_error) ResultCounter.ResultCounterMonad.t
val anf_and_lift_program : Ast.construction list -> (aconstruction list, anf_error) ResultCounter.ResultCounterMonad.t
val pp_anf_error : Stdlib.Format.formatter -> anf_error -> unit
