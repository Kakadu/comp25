type immexpr =
  1. | ImmNum of int
  2. | ImmId of Ast.ident
type binop =
  1. | Add
  2. | Sub
  3. | Mul
  4. | Le
  5. | Lt
  6. | Eq
  7. | Neq
type cexpr =
  1. | CImm of immexpr
  2. | CBinop of binop * immexpr * immexpr
  3. | CApp of immexpr * immexpr list
  4. | CIte of immexpr * aexpr * aexpr
  5. | CFun of Ast.ident * aexpr
and aexpr =
  1. | ACE of cexpr
  2. | ALet of Ast.Expression.rec_flag * Ast.ident * cexpr * aexpr
type astructure_item =
  1. | AStr_value of Ast.Expression.rec_flag * Ast.ident * aexpr
  2. | AStr_eval of aexpr
type aprogram = astructure_item list
val show_aprogram : aprogram -> string
val anf_transform : Ast.program -> aprogram
