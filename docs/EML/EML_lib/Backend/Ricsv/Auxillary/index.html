type call_style =
  1. | Nullary of Base.string
  2. | Curry_chain of {
    1. fname : Base.string;
    2. arity : Base.int;
    3. first_args : Middleend.Anf.immediate Base.list;
    4. rest_args : Middleend.Anf.immediate Base.list;
    }
  3. | Direct of {
    1. fname : Base.string;
    2. args : Middleend.Anf.immediate Base.list;
    }
  4. | Via_apply_nargs of {
    1. fname : Base.string;
    2. nargs : Base.int;
    3. args : Middleend.Anf.immediate Base.list;
    }
val classify_call : nargs:Base.int -> callee_arity_opt:Base.int option -> fname:Base.string -> args:Middleend.Anf.immediate Base.List.t -> call_style
