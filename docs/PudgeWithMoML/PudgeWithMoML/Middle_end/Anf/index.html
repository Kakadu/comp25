Module Middle_end.Anf

type imm =
  1. | ImmConst of Frontend.Ast.literal
  2. | ImmVar of string
type cexpr =
  1. | CImm of imm
  2. | CTuple of imm * imm * imm list
  3. | CBinop of string * imm * imm
  4. | CNot of imm
  5. | CLambda of string * aexpr
  6. | CApp of imm * imm
  7. | CIte of imm * aexpr * aexpr
and aexpr =
  1. | ALet of Frontend.Ast.is_recursive * string * cexpr * aexpr
  2. | ACExpr of cexpr
type binding = Frontend.Ast.ident * aexpr
val pp_binding : Ppx_deriving_runtime.Format.formatter -> binding -> Ppx_deriving_runtime.unit
val show_binding : binding -> Ppx_deriving_runtime.string
type astr_item = Frontend.Ast.is_recursive * binding * binding list
val pp_astr_item : Ppx_deriving_runtime.Format.formatter -> astr_item -> Ppx_deriving_runtime.unit
val show_astr_item : astr_item -> Ppx_deriving_runtime.string
type aprogram = astr_item list
val pp_aprogram : Ppx_deriving_runtime.Format.formatter -> aprogram -> Ppx_deriving_runtime.unit
val show_aprogram : aprogram -> Ppx_deriving_runtime.string
val anf_str_item : Frontend.Ast.structure_item -> astr_item Common.Monad.Counter.t
val anf_program : Frontend.Ast.structure_item list -> aprogram
